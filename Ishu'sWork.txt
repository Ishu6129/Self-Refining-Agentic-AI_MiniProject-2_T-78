File: backend-node/src/agents/planner.agent.ts

import { Groq } from 'groq-sdk';
import { IAgent, AgentInput, AgentOutput, NodeConfig } from './base.agent';

export interface PlannerInput extends AgentInput {
  goal: string;
  context?: Record<string, any>;
}

export interface PlannerOutput extends AgentOutput {
  plan: {
    subtasks: Array<{
      id: string;
      description: string;
      complexity: 'low' | 'medium' | 'high';
      dependencies: string[];
      estimatedTime: string;
    }>;
    totalSteps: number;
    reasoning: string;
  };
}

export class PlannerAgent implements IAgent<PlannerInput, PlannerOutput> {
  private groq: Groq;
  private model: string;
  
  constructor() {
    this.groq = new Groq({ apiKey: process.env.GROQ_API_KEY });
    this.model = process.env.GROQ_MODEL || 'llama-3.3-70b-versatile';
  }

  async execute(input: PlannerInput, config?: NodeConfig): Promise<PlannerOutput> {
    const systemPrompt = `You are a Planner Agent in an Agentic AI system. 
Your job is to decompose high-level goals into structured, executable sub-tasks.

Rules:
1. Create 3-7 sub-tasks maximum (manageable chunks)
2. Order logically - dependencies first
3. Each sub-task must be specific, measurable, and actionable
4. Assign complexity: low (simple), medium (moderate), high (complex)
5. Include estimated time for each sub-task

Respond ONLY in valid JSON format:
{
  "reasoning": "brief explanation of your approach",
  "plan": {
    "subtasks": [
      {
        "id": "task_1",
        "description": "specific action description",
        "complexity": "low|medium|high",
        "dependencies": [],
        "estimatedTime": "30 minutes"
      }
    ],
    "totalSteps": 5
  }
}`;

    try {
      const response = await this.groq.chat.completions.create({
        model: this.model,
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user', content: `Decompose this goal: ${input.goal}` }
        ],
        temperature: 0.3,
        max_tokens: 4096,
        response_format: { type: 'json_object' }
      });

      const result = JSON.parse(response.choices[0].message.content);
      
      return {
        success: true,
        data: result.plan,
        metadata: {
          tokensUsed: response.usage?.total_tokens || 0,
          model: this.model,
          reasoning: result.reasoning
        }
      };
    } catch (error) {
      return {
        success: false,
        error: error.message,
        data: null
      };
    }
  }

  getNodeDefinition() {
    return {
      type: 'planner',
      name: 'Planner',
      description: 'Decomposes goals into sub-tasks',
      icon: '🎯',
      category: 'Core',
      inputs: ['goal'],
      outputs: ['plan'],
      color: '#F59E0B' // Amber-500
    };
  }
}

import { Groq } from 'groq-sdk';
import { IAgent, AgentInput, AgentOutput } from './base.agent';

export interface CriticInput extends AgentInput {
  originalGoal: string;
  executionResult: any;
  executionSteps: any[];
}

export interface CriticOutput extends AgentOutput {
  critique: {
    isSatisfactory: boolean;
    qualityScore: number; // 0-100
    issuesFound: string[];
    strengths: string[];
    improvementSuggestions: string[];
    needsRefinement: boolean;
    refinementFocus: string;
  };
}

export class CriticAgent implements IAgent<CriticInput, CriticOutput> {
  private groq: Groq;
  private model: string;

  constructor() {
    this.groq = new Groq({ apiKey: process.env.GROQ_API_KEY });
    this.model = process.env.GROQ_MODEL || 'llama-3.3-70b-versatile';
  }

  async execute(input: CriticInput): Promise<CriticOutput> {
    const systemPrompt = `You are a Critic Agent in an Agentic AI system.
Your job is to evaluate execution outputs for quality, accuracy, and completeness.

Evaluation Criteria:
1. Factual accuracy (no hallucinations)
2. Completeness (all requirements met)
3. Logical consistency
4. Clarity and coherence
5. Efficiency of solution

Scoring:
- 90-100: Excellent (minor polish needed)
- 70-89: Good (some improvements needed)
- 50-69: Acceptable (significant issues)
- Below 50: Unsatisfactory (needs major rework)

Respond ONLY in valid JSON:
{
  "critique": {
    "isSatisfactory": true/false,
    "qualityScore": 85,
    "issuesFound": ["issue 1", "issue 2"],
    "strengths": ["good aspect 1"],
    "improvementSuggestions": ["suggestion 1"],
    "needsRefinement": true/false,
    "refinementFocus": "specific area to improve"
  }
}`;

    const context = `
Original Goal: ${input.originalGoal}
Execution Steps: ${JSON.stringify(input.executionSteps, null, 2)}
Execution Result: ${JSON.stringify(input.executionResult, null, 2)}
`;

    try {
      const response = await this.groq.chat.completions.create({
        model: this.model,
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user', content: context }
        ],
        temperature: 0.2,
        max_tokens: 4096,
        response_format: { type: 'json_object' }
      });

      const result = JSON.parse(response.choices[0].message.content);

      return {
        success: true,
        data: result.critique,
        metadata: {
          tokensUsed: response.usage?.total_tokens || 0,
          model: this.model
        }
      };
    } catch (error) {
      return {
        success: false,
        error: error.message,
        data: null
      };
    }
  }

  getNodeDefinition() {
    return {
      type: 'critic',
      name: 'Critic',
      description: 'Evaluates execution quality and suggests improvements',
      icon: '🔍',
      category: 'Core',
      inputs: ['originalGoal', 'executionResult', 'executionSteps'],
      outputs: ['critique', 'feedback'],
      color: '#EF4444' // Red-500
    };
  }
}

File: backend-node/src/workflows/workflow.engine.ts

import { PlannerAgent } from '../agents/planner.agent';
import { CriticAgent } from '../agents/critic.agent';
import { ExecutionModel } from '../models/execution.model';
import { NodeStateModel } from '../models/node-state.model';

interface WorkflowNode {
  id: string;
  type: 'planner' | 'executor' | 'critic' | 'custom';
  data: any;
  position: { x: number; y: number };
  connections: string[]; // Next node IDs
}

interface WorkflowExecution {
  workflowId: string;
  executionId: string;
  nodes: WorkflowNode[];
  startNode: string;
  context: Record<string, any>;
}

export class WorkflowEngine {
  private plannerAgent: PlannerAgent;
  private criticAgent: CriticAgent;
  private activeExecutions: Map<string, any>;

  constructor() {
    this.plannerAgent = new PlannerAgent();
    this.criticAgent = new CriticAgent();
    this.activeExecutions = new Map();
  }

  async executeWorkflow(execution: WorkflowExecution): Promise<void> {
    const { executionId, nodes, startNode, context } = execution;
    
    // Initialize execution tracking
    await ExecutionModel.create({
      executionId,
      workflowId: execution.workflowId,
      status: 'running',
      startedAt: new Date(),
      context
    });

    this.activeExecutions.set(executionId, {
      currentNode: startNode,
      context,
      results: {}
    });

    try {
      await this.executeNode(executionId, startNode, nodes);
      
      await ExecutionModel.findOneAndUpdate(
        { executionId },
        { status: 'completed', completedAt: new Date() }
      );
    } catch (error) {
      await ExecutionModel.findOneAndUpdate(
        { executionId },
        { status: 'failed', error: error.message }
      );
      throw error;
    }
  }

  private async executeNode(
    executionId: string, 
    nodeId: string, 
    nodes: WorkflowNode[]
  ): Promise<void> {
    const node = nodes.find(n => n.id === nodeId);
    if (!node) throw new Error(`Node ${nodeId} not found`);

    const execution = this.activeExecutions.get(executionId);
    
    // Update node status to running
    await this.updateNodeState(executionId, nodeId, 'running');

    let result: any;

    try {
      switch (node.type) {
        case 'planner':
          result = await this.plannerAgent.execute({
            goal: execution.context.goal,
            ...node.data
          });
          execution.results.plan = result.data;
          break;

        case 'critic':
          result = await this.criticAgent.execute({
            originalGoal: execution.context.goal,
            executionResult: execution.results.execution,
            executionSteps: execution.results.plan?.subtasks || []
          });
          execution.results.critique = result.data;
          break;

        case 'executor':
          // Call Aryan's Executor service
          result = await this.callExecutorService(node.data, execution.results.plan);
          execution.results.execution = result;
          break;

        default:
          throw new Error(`Unknown node type: ${node.type}`);
      }

      // Mark node as completed
      await this.updateNodeState(executionId, nodeId, 'completed', result);

      // Execute next nodes
      for (const nextNodeId of node.connections) {
        await this.executeNode(executionId, nextNodeId, nodes);
      }

    } catch (error) {
      await this.updateNodeState(executionId, nodeId, 'failed', null, error.message);
      throw error;
    }
  }

  private async updateNodeState(
    executionId: string,
    nodeId: string,
    status: 'running' | 'completed' | 'failed',
    result?: any,
    error?: string
  ): Promise<void> {
    await NodeStateModel.findOneAndUpdate(
      { executionId, nodeId },
      {
        status,
        result,
        error,
        updatedAt: new Date(),
        ...(status === 'running' && { startedAt: new Date() }),
        ...(status === 'completed' && { completedAt: new Date() })
      },
      { upsert: true }
    );

    // Emit real-time update via Socket.io
    // io.emit(`execution:${executionId}`, { nodeId, status, result });
  }

  private async callExecutorService(data: any, plan: any): Promise<any> {
    // HTTP call to Aryan's Flask service or internal Executor
    const response = await fetch('http://localhost:5001/execute', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ data, plan })
    });
    return await response.json();
  }
}

import mongoose, { Schema, Document } from 'mongoose';

export interface IWorkflow extends Document {
  workflowId: string;
  name: string;
  description?: string;
  nodes: Array<{
    id: string;
    type: string;
    position: { x: number; y: number };
    data: any;
  }>;
  connections: Array<{
    source: string;
    target: string;
    sourceHandle?: string;
    targetHandle?: string;
  }>;
  createdBy: string;
  createdAt: Date;
  updatedAt: Date;
}

const WorkflowSchema: Schema = new Schema({
  workflowId: { type: String, required: true, unique: true },
  name: { type: String, required: true },
  description: { type: String },
  nodes: [{
    id: String,
    type: String,
    position: {
      x: Number,
      y: Number
    },
    data: Schema.Types.Mixed
  }],
  connections: [{
    source: String,
    target: String,
    sourceHandle: String,
    targetHandle: String
  }],
  createdBy: { type: String, required: true },
  createdAt: { type: Date, default: Date.now },
  updatedAt: { type: Date, default: Date.now }
});

export default mongoose.model<IWorkflow>('Workflow', WorkflowSchema);
