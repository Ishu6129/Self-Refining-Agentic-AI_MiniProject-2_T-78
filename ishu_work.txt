AGENT 1: PLANNER MODULE (Task Decomposition)
════════════════════════════════════════════════════════════════════════════════
Purpose: Convert high-level goals into structured sub-tasks

Create file: planner_agent.py

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

import os
import json
from groq import Groq
from dotenv import load_dotenv

load_dotenv()

class PlannerAgent:
    def __init__(self):
        self.client = Groq(api_key=os.getenv("GROQ_API_KEY"))
        self.model = os.getenv("GROQ_MODEL")
    
    def decompose_goal(self, goal: str) -> dict:
        \"\"\"Break down goal into sub-tasks\"\"\"
        
        system_prompt = \"\"\"You are a Planner Agent. Your job is to decompose 
high-level goals into specific, actionable sub-tasks.
 
Rules:
1. Create 3-7 sub-tasks maximum
2. Order them logically (dependencies first)
3. Each sub-task must be specific and measurable
4. Include estimated complexity (Low/Medium/High)

Respond ONLY in valid JSON format:
{
    "goal": "original goal",
    "subtasks": [
        {
            "id": 1,
            "description": "specific action",
            "complexity": "Low|Medium|High",
            "dependencies": []
        }
    ],
    "total_steps": 5,
    "estimated_time": "30 minutes"
}\"\"\"

        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": f"Decompose this goal: {goal}"}
            ],
            temperature=0.3,
            max_tokens=2048,
            response_format={"type": "json_object"}
        )
        
        return json.loads(response.choices[0].message.content)

# Test it
if __name__ == "__main__":
    planner = PlannerAgent()
    result = planner.decompose_goal("Build a website for my portfolio")
    print(json.dumps(result, indent=2))

AGENT 2: CRITIC MODULE (Output Evaluation)
════════════════════════════════════════════════════════════════════════════════
Purpose: Review outputs and provide improvement feedback

Create file: critic_agent.py

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

import os
import json
from groq import Groq
from dotenv import load_dotenv

load_dotenv()

class CriticAgent:
    def __init__(self):
        self.client = Groq(api_key=os.getenv("GROQ_API_KEY"))
        self.model = os.getenv("GROQ_MODEL")
    
    def evaluate_output(self, original_goal: str, output: str, 
                       execution_steps: list) -> dict:
        \"\"\"Critique the execution output\"\"\"
        
        system_prompt = \"\"\"You are a Critic Agent. Your job is to evaluate 
AI-generated outputs for quality, accuracy, and completeness.

Check for:
1. Factual errors or hallucinations
2. Missing information or gaps
3. Logical inconsistencies
4. Areas for improvement
5. Whether the goal was fully achieved

Respond ONLY in valid JSON:
{
    "is_satisfactory": true|false,
    "quality_score": 0-100,
    "issues_found": ["issue1", "issue2"],
    "strengths": ["good point1"],
    "improvement_suggestions": ["suggestion1"],
    "needs_refinement": true|false,
    "refinement_focus": "specific area to improve"
}\"\"\"

        context = f\"\"\"
Original Goal: {original_goal}
Execution Steps: {json.dumps(execution_steps)}
Output to Evaluate: {output}
\"\"\"

        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": context}
            ],
            temperature=0.2,
            max_tokens=2048,
            response_format={"type": "json_object"}
        )
        
        return json.loads(response.choices[0].message.content)

# Test it
if __name__ == "__main__":
    critic = CriticAgent()
    result = critic.evaluate_output(
        "Build a calculator app",
        "Created a Python script with add/subtract functions",
        ["Step 1: Write code", "Step 2: Test"]
    )
    print(json.dumps(result, indent=2))
